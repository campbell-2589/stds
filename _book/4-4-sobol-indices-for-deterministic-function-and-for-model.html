4.4 Sobol indices for deterministic function and for model

y.fun <- function(X) {
  
  X1<- X[,1]
  X2<- X[,2]
  X3<- X[,3]
  
  X1+X2+X3
}

yhat.fun<-function(X,lm)
{
  X1<- X[,1]
  X2<- X[,2]
  X3<- X[,3]
  
  yhat <- predict(lm.fit,data.frame(X1=X1,X2=X2,X3=X3))
  return(yhat)
}

nboot = 1000

x <- sobol(model = y.fun, X[1:50,], X[51:100,], order = 3, nboot = nboot)
S.sobol <- x$S
pander(S.sobol)
